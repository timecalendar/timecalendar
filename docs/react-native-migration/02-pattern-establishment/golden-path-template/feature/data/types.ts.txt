// TODO: replace with the feature's domain shape (small, screen-facing — not the DTO).
export interface FeatureItem {
  id: string
  name: string
}

// TODO: replace with the real parse from a raw DTO / storage row. Keep it TOTAL —
// return undefined (or a safe default) for anything malformed; never throw.
export function parseFeatureItem(raw: unknown): FeatureItem | undefined {
  if (
    typeof raw !== "object" ||
    raw === null ||
    typeof (raw as { id?: unknown }).id !== "string" ||
    typeof (raw as { name?: unknown }).name !== "string"
  ) {
    return undefined
  }
  const { id, name } = raw as { id: string; name: string }
  return { id, name }
}
